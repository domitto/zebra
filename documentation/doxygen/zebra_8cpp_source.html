00503                                                 }
00504                                                         ptr += incr;
00505                                                 } else {
00506                                                         epicsStrnEscapedFromRaw(escapedbuff, NBUFF, rxBuffer,
00507                                                                         strlen(rxBuffer));
00508                                                         asynPrint(this->pasynUserSelf, ASYN_TRACE_ERROR,
00509                                                                         "%s:%s: Bad interrupt on encoder %d in '%s'\n", driverName, functionName, a+1, escapedbuff);
00510                                                         break;
00511                                                 }
00512                                         }
00513                                         // publish value to double param
00514                                         setDoubleParam(zebraCapLast[a], dvalue-1);
00515                                         setDoubleParam(zebraCapLast[a], dvalue);
00516                                         // publish value to waveform if we have room
00517                                         if (this->currPt < this->maxPts) {
00518                                                 this->capArrays[a][this->currPt] = dvalue;
00519                                         }
00520                                         // Note: don't do callParamCallbacks here, or we'll swamp asyn
00521                                 }
00522                                 // sanity check
00523                                 if (ptr[0] != '\0') {
00524                                         epicsStrnEscapedFromRaw(escapedbuff, NBUFF, ptr,
00525                                                         strlen(ptr));
00526                                         asynPrint(this->pasynUserSelf, ASYN_TRACE_ERROR,
00527                                                         "%s:%s: Characters remaining in interrupt: '%s'\n", driverName, functionName, escapedbuff);
00528                                 }
00529                                 // advance the counter if allowed
00530                                 if (this->currPt < this->maxPts) {
00531                                         this->currPt++;
00532                                 }
00533                         }
00534                         free(rxBuffer);
00535                 }
00536                 // Update any params we have got, this means that the max update rate
00537                 // of the waveform last values is this loop tick (10Hz).
00538                 callParamCallbacks();
00539                 this->unlock();
00540                 // Work out how long to sleep for so each loop iteration takes 0.1s
00541                 epicsTimeGetCurrent(&end);
00542                 double timeToSleep = 0.1 - epicsTimeDiffInSeconds(&end, &start);
00543                 if (timeToSleep > 0) {
00544                         epicsThreadSleep(timeToSleep);
00545                 } else {
00546                         // Got to sleep for a bit in case something else is waiting for the lock
00547                         epicsThreadSleep(0.01);
00548                         //printf("Not enough time to poll properly %f\n", timeToSleep);
00549                 }
00550         }
00551 }

00553 /* This is the function that will be run for the poll thread */
00554 void zebra::pollTask() {
00555         const char *functionName = "pollTask";
00556         int value, caploparam, caphiparam, lastcap, downloading;
00557         double loopTime;
00558         const reg *r;
00559         unsigned int sys, poll = 0, iteration = 0;
00560         char *rxBuffer, escapedbuff[NBUFF];
00561         epicsTimeStamp start, end;
00562         asynStatus status = asynSuccess;
00563         findParam("PC_NUM_CAPLO", &caploparam);
00564         findParam("PC_NUM_CAPHI", &caphiparam);
00565         // Wait 1 second until port is up
00566         epicsThreadSleep(1.0);
00567         while (true) {
00568                 // alternate between the next slow reg, and all the fast regs
00569                 epicsTimeGetCurrent(&start);
00570                 this->lock();
00571                 // If there are any responses on the queue they must be junk
00572                 while (epicsMessageQueuePending(this->msgQId)) {
00573                         epicsMessageQueueReceive(this->msgQId, &rxBuffer,
00574                                         sizeof(&rxBuffer));
00575                         epicsStrnEscapedFromRaw(escapedbuff, NBUFF, rxBuffer,
00576                                         strlen(rxBuffer));
00577                         asynPrint(this->pasynUserSelf, ASYN_TRACE_ERROR,
00578                                         "%s:%s: Junk message in buffer '%s'\n", driverName, functionName, escapedbuff);
00579                         status = asynError;
00580                         free(rxBuffer);
00581                 }
00582                 // Work out if we are currently downloading
00583                 getIntegerParam(zebraArrayAcq, &downloading);
00584                 // Check what PC_NUM_CAPLO is now so can check if it rolled over
00585                 getIntegerParam(caploparam, &lastcap);
00586                 if (downloading) {
00587                         // If we are downloading, then must wait for responses as FPGA is heavily loaded
00588                         for (sys = NREGS - FASTREGS; sys < NREGS; sys++) {
00589                                 // Send demand to zebra
00590                                 this->getReg(&(reg_lookup[sys]), &value);
00591                         }
00592                         // Now wait a second until we do it again
00593                         loopTime = 1.0;
00594                 } else if (iteration == 0) {
00595                         // First send requests for all the system
00596                         for (sys = NREGS - FASTREGS; sys < NREGS; sys++) {
00597                                 // Send demand to zebra
00598                                 this->sendGetReg(&(reg_lookup[sys]));
00599                                 epicsThreadSleep(DELAYMULTIREAD);
00600                         }
00601                         // Now get values back
00602                         for (sys = NREGS - FASTREGS; sys < NREGS; sys++) {
00603                                 // wait for a response on the message queue
00604                                 this->receiveGetReg(&(reg_lookup[sys]), &value);
00605                         }
00606                         // Now wait 0.25 seconds until we get the regs below
00607                         loopTime = 0.25;
00608                 } else {
00609                         // Get the register value from zebra
00610                         status = this->getReg(&(reg_lookup[poll]), &value);
00611                         // skip one, then keep on skipping until it isn't a command
00612                         for (r = NULL; r == NULL || r->type == regCmd;) {
00613                                 // Move to next register
00614                                 if (++poll >= NREGS - FASTREGS) {
00615                                         poll = 0;
00616                                         // Done one complete cycle so write to file allowed.
00617                                         this->doneInit = 1;
00618                                 }
00619                                 r = &(reg_lookup[poll]);
00620                         }
00621                         // Now wait 0.25 seconds until the next poll
00622                         loopTime = 0.25;
00623                 }
00624                 // check what NUM_CAP is now
00625                 getIntegerParam(caploparam, &value);
00626                 // If this is PC_NUM_CAPLO and it has rolled over, then trigger a PC_NUM_CAP_HI update
00627                 if (value < lastcap) {
00628                         //printf("Rollover!\n");
00629                         this->getReg(PARAM2REG(caphiparam), &value);
00630                 }
00631 
00632                 // Iteration 0 is all 6 FASTREGS, iterations 1-3 are the next slow polled regs
00633                 iteration++;
00634                 if (iteration > 3) iteration = 0;
00635                 // Update params
00636                 callParamCallbacks();
00637                 this->unlock();
00638                 // We try to run this loop at 4Hz so that system values get done at 1Hz
00639                 // or at 1Hz during download time
00640                 epicsTimeGetCurrent(&end);
00641                 double timeToSleep = loopTime - epicsTimeDiffInSeconds(&end, &start);
00642                 if (timeToSleep > 0) {
00643                         epicsThreadSleep(timeToSleep);
00644                 } else {
00645                         // Got to sleep for a bit in case something else is waiting for the lock
00646                         epicsThreadSleep(0.01);
00647                         //printf("Not enough time to poll properly %f\n", timeToSleep);
00648                 }
00649         }
00650 }

00969 asynStatus zebra::writeInt32(asynUser *pasynUser, epicsInt32 value) {

00971         asynStatus status = asynError;

00973 
00974         /* Any work we need to do */
00975         int param = pasynUser->reason;

00995                 status = this->flashCmd("L");

00997                 char fileName[NBUFF];

01000                         status = this->configRead(fileName);
01001                 } else {

01013                 value = value % NSYSBUS;

01016                 // Resend all the waveforms as we have changed the filter
01017                 setIntegerParam(zebraNumDown, 0);

01019         }

01024 /* This function calls back on the time and position waveform values
01025  called with the lock taken */
01026 asynStatus zebra::callbackWaveforms() {
01027         int sel, lastUpdatePt;
01028         double *src;
01029         getIntegerParam(zebraNumDown, &lastUpdatePt);
01030         if (lastUpdatePt != this->currPt) {
01031                 // printf("Update %d %d\n", this->lastUpdatePt, this->currPt);
01032                 // store the last update so we don't get repeated updates
01033                 setIntegerParam(zebraNumDown, this->currPt);
01034 
01035                 if (this->currPt < this->maxPts) {
01036                         // horrible hack for edm plotting
01037                         // set the last+1 time point to be the same as the last, this
01038                         // means the last point on the time/pos plot is (last, 0), which
01039                         // gives a straight line back to (0,0) without confusing the user
01040                         this->PCTime[this->currPt] = this->PCTime[this->currPt - 1];
01041                         doCallbacksFloat64Array(this->PCTime, this->currPt + 1, zebraPCTime, 0);
01042                 } else {
01043                         doCallbacksFloat64Array(this->PCTime, this->currPt, zebraPCTime, 0);
01044                 }
01045 
01046                 /* Filter the relevant sys_bus array with filtSel[a] and put the value in filtArray[a] */
01047                 for (int a = 0; a < NFILT; a++) {
01048                         getIntegerParam(zebraFiltSel[a], &sel);
01049                         if (sel < 32) {
01050                                 src = this->capArrays[4]; // SYS_BUS1
01051                         } else {
01052                                 src = this->capArrays[5]; // SYS_BUS2
01053                                 sel -= 32;
01054                         }
01055                         for (int i = 0; i < this->maxPts; i++) {
01056                                 this->filtArrays[a][i] = (((unsigned int)(src[i]+0.5)) >> sel) & 1;                                
01057                         }
01058                         doCallbacksInt8Array(this->filtArrays[a], this->currPt,
01059                                         zebraFiltArrays[a], 0);
01060                 }
01061 
01062                 // update capture arrays
01063                 for (int a = 0; a < NARRAYS; a++) {
01064                         doCallbacksFloat64Array(this->capArrays[a], this->currPt,
01065                                         zebraCapArrays[a], 0);
01066                 }
01067
